model Cabin {
  id String @id @default(cuid())
  name String
  description String?
  capacity Int
  weekdayPrice Int
  weekendPrice Int
  extraPersonPrice Int
  images String[]
  bookings Booking[]
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
}
model Booking {
  id String @id @default(cuid())
  cabinId String
  cabin Cabin @relation(fields:[cabinId], references:[id])
  guestName String
  guestEmail String
  guestPhone String?
  guests Int
  checkIn DateTime
  checkOut DateTime
  totalPrice Int?
  status String @default("confirmed")
  source String?
  notes String?
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
  @@index([cabinId, checkIn, checkOut])
}
model Review { id String @id @default(cuid()) name String rating Int comment String createdAt DateTime @default(now()) }
model Gallery { id String @id @default(cuid()) url String alt String? order Int @default(0) }


Scripts:

{ "scripts": {
  "db:generate": "prisma generate",
  "db:migrate": "prisma migrate deploy",
  "db:seed": "tsx scripts/seed.ts"
}}


Seed mínimo (idempotente) en scripts/seed.ts:

import { PrismaClient } from "@prisma/client";
const db = new PrismaClient();
async function main(){
  await db.cabin.upsert({
    where:{ id:"default-cabin" }, update:{},
    create:{
      id:"default-cabin",
      name:"Cabaña Vista Montaña",
      description:"Cálida, con desayuno incluido.",
      capacity:3,
      weekdayPrice:350000,
      weekendPrice:450000,
      extraPersonPrice:50000,
      images:["/hero.jpg","/tarifas.jpg"]
    }
  });
}
main().finally(()=>db.$disconnect());


Ejecuta (no destructivo):

npm run db:generate && npm run db:migrate && npm run db:seed
